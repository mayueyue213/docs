    let deepClone = function func(obj) {
  if (typeof obj === 'object') {
    const result = Array.isArray(obj) ? [] : {};
    for (const key in obj) {
      result[key] = typeof obj[key] === 'object' ? func(obj[key]) : obj[key];
    }

    return result;
  }

  return obj;
}

    # 防抖和节流

    防抖(debounce)和节流(throttle)都是性能优化策略，都是为了减少在一定时间内，同一处理函数的执行次数。

    # debounce

    在一定时间内，重复发生的事件只处理一次，避免重复执行。比如

    • scroll、resize事件：这类连续性事件，往往都需要重新获取尺寸信息，并且涉及一些计算
    • 实时输入处理：比如 element 的 autocomplete 组件，既然需要实时输入远程查询，这个“实时”，自然就是需要防抖的；

    一个基础的防抖函数如下

    const debounce = (delay = 100, func, immediate) => {
  let timer = 0;
  let self = this;
  return function(...args) {
    if (timer) {
      return;
    }

    timer = setTimeout(function() {
      func.apply(self, args);
      timer = 0;
    }, delay)
  }
}

    # throttle

    节流的核心思想是，如果距离上一次执行还没有相隔 delay，那就等时间到了再执行一次。

    const throttle = (delay = 100, func) => {
  let timer = 0;
  let lastExec = 0;
  let self = this;

  return function(...args) {
    timer && clearTimeout(timer);
    let elapsed = Date.now() - lastExec;

    if (elapsed > delay) {
      lastExec = Date.now();
      func.apply(self, args);
    } else {
      timer = setTimeout(function() {
        lastExec = Date.now();
        func.apply(self, args);
      }, delay - elapsed);
    }
  }
}

    防抖和节流都是在 delay 时间段内，只能执行一次，区别在于

    • 防抖：在 delay 时间段内，发生 n 次，执行第 1 次事件的处理，下一次执行的是下一个 delay 时间段内发生的第一次处理，即 n + 1；

    • 节流：在 delay 时间段内，发生 n 次，执行第 1 次事件的处理，下一次执行的是第 n 次事件的处理；

    对于这类工具函数，应使用成熟统一的方案，可以参考 throttle-debounce (opens new window)。

    # 图片压缩

    一种图片压缩的方式是将大图片绘制到较小的 canvas 画布上，然后再生成小的突破。知识点

    • canvas 基本操作
    • canvas.toDataURL
    // 只是一个过程演示，就不关注写法和质量了。
const fileEl = document.getElementById('file');
fileEl.addEventListener('change', function (event) {
  const file = event.target.files[0];
  const url = URL.createObjectURL(event.target.files[0]);

  const img = new Image();
  img.src = url;
  document.body.appendChild(img);
  img.addEventListener('load', function () {
    const canvas = document.createElement('canvas');
    const ctx = canvas.getContext('2d');
    canvas.width = 200;
    canvas.height = 200;
    ctx.drawImage(img, 0, 0, 200, 200);
    document.body.appendChild(canvas)
  })
});

    # 柯里化

    柯里化是函数式变成里面的概念，每次调用传递一部分参数给函数，返回一个函数去处理剩余参数。 最常见的用来介绍柯里化概念的题目是下面这个

    function originAdd(a, b, c) {
  return a + b + c;
}

add(1)(2)(3);
add(1, 2, 3);
add(1, 2)(3);

    思路其实也并不复杂，每次调用判断参数个数是否达到要求个数，符合就调用，不符合就返回一个函数继续接收参数。 一种实现如下

    function curry(fn) {
  let _args = [];
  return function foo(...args) {
    _args = _args.concat(args);
    if (_args.length >= fn.length) {
      args = [..._args]
      _args = [];
      return fn.apply(fn, args);
    }
    
    return foo;
  }
}

const add = curry(originAdd);
